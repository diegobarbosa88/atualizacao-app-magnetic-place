If it does, use `import`. If not (package.json has no `"type"` field or `"type": "commonjs"`), use CommonJS (`require`/`module.exports`). Do NOT use `require` with ESM or `import` with CJS.

    NOTE: Check `cat package.json | grep '"type"'` before writing the file. If `"type": "module"` is present, use ESM. Otherwise use CommonJS.

    Implement the following logic exactly:

    ```javascript
    // ESM version (use if package.json has "type": "module")
    import { google } from 'googleapis';
    import { createClient } from '@supabase/supabase-js';

    // Helper: recursively find PDF/XML attachment parts in a MIME tree
    // Multipart emails nest attachments inside multipart/mixed or multipart/alternative parts.
    // Only checking top-level parts misses attachments. Always recurse.
    function findAttachmentParts(parts) {
      const results = [];
      if (!parts || !Array.isArray(parts)) return results;
      for (const part of parts) {
        if (
          (part.mimeType === 'application/pdf' || part.mimeType === 'application/xml') &&
          part.body?.attachmentId
        ) {
          const rawName = part.filename || `attachment_${Date.now()}.${part.mimeType === 'application/pdf' ? 'pdf' : 'xml'}`;
          // Sanitise filename: strip path traversal and non-safe chars (per RESEARCH.md security)
          const filename = rawName.replace(/\.\.[/\\]/g, '').replace(/[^a-zA-Z0-9._\-\s]/g, '_');
          results.push({
            attachmentId: part.body.attachmentId,
            filename,
            mimeType: part.mimeType,
          });
        }
        // Recurse into nested parts
        if (part.parts && part.parts.length > 0) {
          results.push(...findAttachmentParts(part.parts));
        }
      }
      return results;
    }

    export default async function handler(req, res) {
      // Method guard
      if (req.method !== 'POST') {
        return res.status(405).json({ error: 'Method not allowed' });
      }

      // Auth guard (D-05 + RESEARCH.md Security Domain)
      // GMAIL_IMPORT_SECRET is a shared secret stored in Vercel env vars.
      // The admin UI sends it via x-import-secret header using VITE_GMAIL_IMPORT_SECRET.
      if (req.headers['x-import-secret'] !== process.env.GMAIL_IMPORT_SECRET) {
        return res.status(401).json({ error: 'Unauthorized' });
      }

      let processados = 0;
      let ficheiros = 0;
      const erros = [];

      try {
        // 1. Init Gmail client (D-03, D-04)
        // GMAIL_SERVICE_ACCOUNT_JSON is the base64-encoded contents of credentials.json
        // Buffer.from(..., 'base64') handles newlines in private key correctly (RESEARCH Pitfall 2)
        const credentials = JSON.parse(
          Buffer.from(process.env.GMAIL_SERVICE_ACCOUNT_JSON, 'base64').toString('utf-8')
        );
        const auth = new google.auth.JWT({
          email: credentials.client_email,
          key: credentials.private_key,
          scopes: ['https://www.googleapis.com/auth/gmail.modify'],
          subject: process.env.GMAIL_TARGET_EMAIL,
        });
        await auth.authorize();
        const gmail = google.gmail({ version: 'v1', auth });

        // 2. Init Supabase admin client
        // Must use SUPABASE_URL (no VITE_ prefix) + SUPABASE_SERVICE_ROLE_KEY
        // VITE_* vars are NOT available in Vercel Node.js runtime (RESEARCH Pitfall 4)
        const supabaseAdmin = createClient(
          process.env.SUPABASE_URL,
          process.env.SUPABASE_SERVICE_ROLE_KEY
        );

        // 3. List unread emails with invoice attachments (D-06, D-07)
        const listRes = await gmail.users.messages.list({
          userId: 'me',
          q: 'is:unread has:attachment {subject:fatura subject:invoice subject:FT}',
          maxResults: 50,  // MVP cap — no pagination in Phase 17
        });

        const messages = listRes.data.messages || [];

        // 4. Process each email
        for (const msg of messages) {
          try {
            // Get full payload (list only returns id + threadId — RESEARCH Pitfall 6)
            const msgRes = await gmail.users.messages.get({
              userId: 'me',
              id: msg.id,
              format: 'full',
            });

            // Find all PDF/XML attachment parts (recursively — RESEARCH Pattern 3)
            const attachmentParts = findAttachmentParts(msgRes.data.payload?.parts || []);

            for (const part of attachmentParts) {
              // Per D-14: each attachment has its own try/catch (fail-partial)
              try {
                // Download attachment (D-08, D-09)
                const attRes = await gmail.users.messages.attachments.get({
                  userId: 'me',
                  messageId: msg.id,
                  id: part.attachmentId,
                });

                // Base64Url decode — MUST use 'base64url' encoding, not 'base64'
                // Gmail uses URL-safe Base64 (RFC 4648 §5): replaces + with - and / with _
                // 'base64' decoding would silently corrupt these bytes (RESEARCH Don't Hand-Roll)
                const buffer = Buffer.from(attRes.data.data, 'base64url');

                // Upload to Supabase Storage (D-10)
                const storagePath = `faturas/${msg.id}/${part.filename}`;
                const { error: uploadError } = await supabaseAdmin.storage
                  .from('faturas')
                  .upload(storagePath, buffer, {
                    contentType: part.mimeType,
                    upsert: true,  // idempotent re-runs
                  });
                if (uploadError) throw uploadError;

                // Get public URL (D-11)
                const { data: urlData } = supabaseAdmin.storage
                  .from('faturas')
                  .getPublicUrl(storagePath);

                // Record in faturas table (NOT the documents table — per phase spec)
                const { error: dbError } = await supabaseAdmin.from('faturas').insert({
                  gmail_message_id: msg.id,
                  filename: part.filename,
                  storage_path: storagePath,
                  url: urlData.publicUrl,
                  mime_type: part.mimeType,
                  tamanho: buffer.length,
                });
                if (dbError) throw dbError;

                ficheiros++;
              } catch (attachErr) {
                // D-14: isolated failure — record error, continue with next attachment
                erros.push({
                  messageId: msg.id,
                  filename: part.filename,
                  error: attachErr.message,
                });
              }
            }

            // Mark email as read after all attachments processed (D-15, D-16)
            // Done even if some attachments failed (D-13) — prevents infinite reprocessing
            // Note: googleapis v100+ uses 'requestBody', not 'resource' (RESEARCH Pattern 5)
            await gmail.users.messages.modify({
              userId: 'me',
              id: msg.id,
              requestBody: { removeLabelIds: ['UNREAD'] },
            });

            processados++;
          } catch (msgErr) {
            // D-12: one email failing doesn't stop others
            erros.push({ messageId: msg.id, error: msgErr.message });
          }
        }

        // D-02: response format
        return res.status(200).json({ processados, ficheiros, erros });

      } catch (fatalErr) {
        // Fatal: auth failure, Supabase init failure, etc.
        return res.status(500).json({ error: fatalErr.message });
      }
    }
    ```
