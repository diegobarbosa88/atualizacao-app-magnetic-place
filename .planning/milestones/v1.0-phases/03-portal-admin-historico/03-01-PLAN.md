    ```javascript
    // Apply client filter

      historico = historico.filter(d => d.clientId === filterClient);
    }
    
    // Apply status filter (resolved/pending/rejected based on approval status)
    if (filterStatus) {
      historico = historico.filter(d => {
        const approval = clientApprovals?.find(a => a.client_id === d.clientId && a.month === d.month);
        if (filterStatus === 'pending') return !approval || approval.status === 'pending';
        if (filterStatus === 'resolved') return approval?.status === 'approved';
        if (filterStatus === 'rejected') return approval?.status === 'rejected';
        return true;
      });
    }
    
    // Apply month filter
    if (filterMonth) {
      historico = historico.filter(d => d.month === filterMonth);
    }
    ```
    
    Note: Adjust field names based on actual documents table schema. The key point is:
    - filterClient filters by client ID
    - filterStatus filters by approval status (pending/approved/rejected)
    - filterMonth filters by YYYY-MM month string
